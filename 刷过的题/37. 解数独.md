class Solution {
    boolean succ;
    public void solveSudoku(char[][] board) {
        
        backtrack(board,0);
    }

    public boolean backtrack(char[][] board,int row){
        for(int i=row;i<board.length;i++){
            for(int j=0;j<board[0].length;j++){
                if(board[i][j]=='.'){
                    for(int k=1;k<=9;k++){
                        if(isValid(board,k,i,j)){
                            board[i][j]=(char)(k+'0');
                            if(backtrack(board,row))
                                return true;
                            board[i][j]='.';
                        }
                    }
                    return false;
                }
            }
        }
        return true;
    }

    public boolean isValid(char[][] board,int num,int row,int col){
        //判断行是否有效
        for(int i=0;i<9;i++){
            if(board[row][i]-'0' == num)
                return false;
        }

        //判断列是否有效
        for(int i=0;i<9;i++){
            if(board[i][col]-'0' == num)
                return false;
        }

        //判断9宫格是否有效
        int row1=row/3*3;
        int col1=col/3*3;
        for(int i=row1;i<=row1+2;i++){
            for(int j=col1;j<=col1+2;j++){
                if(board[i][j]-'0'==num)
                    return false;
            }
        }
        return true;
    }

    
}
